Build Order

Step 1: Python Runtime
    ↓

    ↓

step by step

cd C:\dev\ksnlabs\ocritd

# Step 1: Build Python runtime (uses wheelhouse for offline install)
.\BUILD.ps1 -BuildPython

# Verify it worked
.\runtime\python\python.exe --version
.\runtime\python\python.exe -m engine.ocritd --check-requirements

# Step 2: Build GUI (requires .NET SDK)
.\BUILD.ps1 -BuildGUI

# Step 3: Build MSI (requires WiX Toolset)
.\BUILD.ps1 -BuildMSI


All at Once
powershell.\BUILD.ps1 -All
